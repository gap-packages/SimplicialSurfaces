1540InstallGlobalFunction( StabilizerOfBlockNC, function(G,B)
1541local S,j;
1542  S:=StabChainOp(G,rec(base:=[B[1]],reduced:=false));
1543  S:=StructuralCopy(S);
1544
1545  # Make <S> the stabilizer of the block <B>.
1546  InsertTrivialStabilizer(S.stabilizer,B[1]);
1547  j := 1;
1548  while                                j < Length( B )
1549 and Length( S.stabilizer.orbit ) < Length( B )  do
1550      j := j + 1;
1551      if IsBound( S.translabels[ B[ j ] ] )  then
1552   AddGeneratorsExtendSchreierTree( S.stabilizer,
1553   [ InverseRepresentative( S, B[ j ] ) ] );
1554      fi;
1555  od;
1556  return GroupStabChain(G,S.stabilizer,true);
1557end );
