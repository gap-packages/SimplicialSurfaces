40#M  \<( <obj1>, <obj2> )  . . . . . . . for two objects in different families
41##
42##  <#GAPDoc Label="[1]{object.gi}">
43##  Only for the following kinds of objects, an ordering via <C>&lt;</C> of objects
44##  in <E>different</E> families (see&nbsp;<Ref Sect="Families"/>) is supported.
45##  Rationals (see&nbsp;<Ref Func="IsRat"/>) are smallest,
46##  next are cyclotomics (see&nbsp;<Ref Func="IsCyclotomic"/>),
47##  followed by finite field elements (see&nbsp;<Ref Func="IsFFE"/>);
48##  finite field elements in different characteristics are compared
49##  via their characteristics,
50##  next are permutations (see&nbsp;<Ref Func="IsPerm"/>),
51##  followed by the boolean values <K>true</K>, <K>false</K>, and <K>fail</K>
52##  (see&nbsp;<Ref Func="IsBool"/>),
53##  characters (such as <C>{</C>}a{'}', see&nbsp;<Ref Func="IsChar"/>),
54##  and lists (see&nbsp;<Ref Func="IsList"/>) are largest;
55##  note that two lists can be compared with <C>&lt;</C> if and only if their
56##  elements are again objects that can be compared with <C>&lt;</C>.
57##  <P/>
58##  For other objects, &GAP; does <E>not</E> provide an ordering via <C>&lt;</C>.
59##  The reason for this is that a total ordering of all &GAP; objects
60##  would be hard to maintain when new kinds of objects are introduced,
61##  and such a total ordering is hardly used in its full generality.
62##  <P/>
63##  However, for objects in the filters listed above, the ordering via <C>&lt;</C>
64##  has turned out to be useful.
65##  For example, one can form <E>sorted lists</E> containing integers and nested
66##  lists of integers, and then search in them using <C>PositionSorted</C>
67##  (see&nbsp;<Ref Sect="Finding Positions in Lists"/>).
68##  <P/>
69##  Of course it would in principle be possible to define an ordering
70##  via <C>&lt;</C> also for certain other objects,
71##  by installing appropriate methods for the operation <C>\&lt;</C>.
72##  But this may lead to problems at least as soon as one loads &GAP; code
73##  in which the same is done, under the assumption that one is completely
74##  free to define an ordering via <C>&lt;</C> for other objects than the ones
75##  for which the <Q>official</Q> &GAP; provides already an ordering via <C>&lt;</C>.
76##  <#/GAPDoc>
77##

687##
688#F  NewObjectMarker( )
689#F  MarkObject( <marks>, <obj> )
690#F  UnmarkObject( <marks>, <obj> )
691#F  ClearObjectMarker( <marks> )
692##  
693##  Utilities to detect identical objects. Used in MemoryUsage below,
694##  but probably of independent interest.
695##  
696
697InstallGlobalFunction( NewObjectMarker, function()
698  local marks, len;
699  marks := rec();
700  len := 2 * MASTER_POINTER_NUMBER(2^100);
701  marks.marks := BlistList([1..len], []);
702  marks.ids := [];
703  # If this is set to some higher values the clearing of the entries
704  # takes more time than creating .marks from scratch.
705  marks.maxids := QuoInt(Length(marks.marks), 30);
706  return marks;
707end);
708
709InstallGlobalFunction( MarkObject, function(marks, obj)
710  local id, res;
711  id := MASTER_POINTER_NUMBER(obj);
712  if id > Length(marks.marks) then
713    marks.marks :=  BlistList( [ 1 .. 2 * id ],
714                                    PositionsTrueBlist(marks.marks));
715  fi;
716  if marks.maxids > Length(marks.ids) then
717    Add(marks.ids, id);
718  fi;
719  res := marks.marks[id];
720  marks.marks[id] := true;
721  return res;
722end);
723
724InstallGlobalFunction(UnmarkObject, function(marks, obj)
725  local id;
726  id := MASTER_POINTER_NUMBER(obj);
727  if id > Length(marks.marks) or not marks.marks[id] then
728    return false;
729  else
730    marks.marks[id] := false;
731    return true;
732  fi;
733end);
734
735InstallGlobalFunction(ClearObjectMarker, function(marks)
736  if Length(marks.ids) < marks.maxids then
737    marks.marks{marks.ids} := BlistList([1..Length(marks.ids)], []);
738  else
739    marks.marks := BlistList([1..Length(marks.marks)], []);
740  fi;
741  marks.ids := [];
742end);

746#M  MemoryUsage( <obj> ) . . . . . . . . . . . . .return fail in general
747##  
748BIND_GLOBAL( "MEMUSAGECACHE", NewObjectMarker( ) );
749MEMUSAGECACHE.depth := 0;
750
751InstallGlobalFunction( MU_AddToCache, function ( obj )
752  return MarkObject(MEMUSAGECACHE, obj);
753end );
754
755InstallGlobalFunction( MU_Finalize, function (  )
756  local mks, i;
757  if MEMUSAGECACHE.depth <= 0  then
758      Error( "MemoryUsage depth has gone below zero!" );
759  fi;
760  MEMUSAGECACHE.depth := MEMUSAGECACHE.depth - 1;
761  if MEMUSAGECACHE.depth = 0  then
762    ClearObjectMarker(MEMUSAGECACHE);
763  fi;
764end );
765
766InstallMethod( MemoryUsage, "fallback method for objs without subobjs",
767  [ IsObject ],
768  function( o )
769    local mem;
770    mem := SHALLOW_SIZE(o);
771    if mem = 0 then 
772        return MU_MemPointer;
773    else
774        # a proper object, thus we have to add it to the database
775        # to not count it again!
776        if not(MU_AddToCache(o)) then
777            mem := mem + MU_MemBagHeader + MU_MemPointer;
778            # This is for the bag, the header, and the master pointer
779        else 
780            mem := 0;   # already counted
781        fi;
782        if MEMUSAGECACHE.depth = 0 then   
783            # we were the first to be called, thus we have to do the cleanup
784            ClearObjectMarker( MEMUSAGECACHE );
785        fi;
786    fi;
787    return mem;
788  end );

790InstallMethod( MemoryUsage, "for a plist",
791  [ IsList and IsPlistRep ],
792  function( o )
793    local mem,known,i;
794    known := MU_AddToCache( o );
795    if known = false then    # not yet known
796        MEMUSAGECACHE.depth := MEMUSAGECACHE.depth + 1;
797        mem := SHALLOW_SIZE(o) + MU_MemBagHeader + MU_MemPointer;
798        # Again the bag, its header, and the master pointer
799        for i in [1..Length(o)] do
800            if IsBound(o[i]) then
801                if SHALLOW_SIZE(o[i]) > 0 then    # a subobject!
802                    mem := mem + MemoryUsage(o[i]);
803                fi;
804            fi;
805        od;
806        MU_Finalize();
807        return mem;
808    fi;
809    return 0;    # already counted
810  end );
