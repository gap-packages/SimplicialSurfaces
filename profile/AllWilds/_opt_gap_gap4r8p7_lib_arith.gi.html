511#M  <int> * <elm>
512##
513##  `PROD_INT_OBJ' is a kernel function that first checks whether <int> is
514##  equal to one of
515##  `0' (then `ZERO' is called),
516##  `1' (then a copy of <elm> is returned), or
517##  `-1' (then `AINV' is called).
518##  Otherwise if <int> is negative then the product of the additive inverses
519##  of <int> and <elm> is computed,
520##  where the product with positive <int> is formed by repeated doubling.
521##
522##  So this method is optimal if <int> is equal to `0',
523##  and generic (i.e., relies only on `ZeroOp', `\+', `AdditiveInverseOp')
524##  otherwise.
525##

527    "positive integer * additive element",

651##
652#M  Order( <obj> )
653##
654InstallMethod( Order,
655    "for a mult. element-with-one",
656    [ IsMultiplicativeElementWithOne ],
657    function( obj )
658    local one, pow, ord;
659
660    # Try to get the identity of the object.
661    one:= One( obj );
662    if one = fail then
663      Error( "<obj> has not an identity" );
664    fi;
665
666    # Check that the object is invertible.
667    if Inverse( obj ) = fail then
668      Error( "<obj> is not invertible" );
669    fi;
670
671    # Compute the order.
672#T warnings about possibly infinite order ??
673    pow:= obj;
674    ord:= 1;
675    while pow <> one do
676    ord:= ord + 1;
677      pow:= pow * obj;
678    od;
679
680    # Return the order.
681    return ord;
682    end );
