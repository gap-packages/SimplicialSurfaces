17##  <Func Name="ProfileLineByLine" Arg="filename[,options]"/>
18##
19##  <Description>
20##  <Ref Func="ProfileLineByLine"/> begins GAP recording profiling
21##  data to the file <A>filename</A>. This file will get *very* large
22##  very quickly. This file is compressed using gzip to reduce its size.
23##
24##  <A>options</A> is an optional dictionary, which sets various
25##  configuration options. These are
26##  <List>
27##  <Mark>coverage</Mark>
28##      <Item> Boolean (defaults to false). If this is enabled,
29##      only information about which lines are read and executed is
30##      stored. Enabling this is the same as calling
31##      <Ref Func="CoverageLineByLine"/>. Using this ignores all other
32##      options.</Item>
33##
34##  <Mark>justStat</Mark>
35##      <Item> Boolean (defaults to false). This switches profiling to only
36##      consider entire statements, rather than parts of statements.
37##      In general this will provide a courser profile, but produce smaller
38##      files.</Item>
39##
40##  <Mark>wallTime</Mark>
41##      <Item> Boolean (defaults to true). Sets if time should be measured
42##      using wall-clock time or CPU time.
43##      (measuring wall-clock time is lower overhead).
44##      </Item>
45##
46##  <Mark>resolution</Mark>
47##      <Item> Integer (defaults to 0). How frequently (in nanoseconds) 
48##      to check which line is being executed. Setting this to a
49##      non-zero value improves performance and produces smaller
50##      traces, at the cost of accuracy. Setting this to a non-zero
51##      value will also make the number of executions per
52##      statement become inaccurance. However,i profiling
53##      will still accurately record which statements
54##      are executed at all.</Item>
55##  </List>
56##  </Description>
57##  </ManSection>
58##  <#/GAPDoc>
59##
60BIND_GLOBAL("ProfileLineByLine",function(arg)
61    local optRec, r;   
62    if Length(arg) = 0 or Length(arg) > 2 or not(IsString(arg[1])) then
63      Error("usage: ProfileLineByLine(filename,[options]");
64    fi;
65
66    optRec := rec(coverage := false,
67                  justStat := false,
68                  wallTime := true, 
69                  resolution := 0);
70    if Length(arg) = 2 then
71      if not(IsRecord(arg[2])) then
72        Error("usage: ProfileLineByLine(filename,[options]");
73      fi;
74      
75      for r in RecNames(arg[2]) do
76        if not(IsBound(optRec.(r))) then
77          Error(Concatenation("Bad option: ", r));
78        fi;
79        optRec.(r) := arg[2].(r);
80      od;
81    fi;
82    
83    if Length(arg[1]) < 3 or 
84       arg[1]{[Length(arg[1])-2..Length(arg[1])]} <> ".gz" then
85      Info(InfoWarning, 1, "Profile filenames must end in .gz to enable compression");
86    fi;
87    return ACTIVATE_PROFILING(arg[1], optRec.coverage, optRec.justStat,
88                                      optRec.wallTime, optRec.resolution);
89end);
