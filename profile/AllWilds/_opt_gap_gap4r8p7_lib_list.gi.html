713##
714#F  IteratorList( <list> )
715##
716##  returns an iterator constructed from the list <list>,
717##  which stores the underlying list in the component `list'
718##  and the current position in the component `pos'.
719##
720##  It may happen that the underlying list is a enumerator of a domain
721##  whose size cannot be computed easily.
722##  In such cases, the methods for `IsDoneIterator' and `NextIterator'
723##  shall avoid calling `Length' for the enumerator.
724##  Therefore a special representation exist for iterators of dense immutable
725##  lists.
726##  (Note that the `Length' call is unavoidable for iterators of non-dense
727##  lists.)
728##
729BindGlobal( "IsDoneIterator_List",
730    iter -> ( iter!.pos = iter!.len ) );
731
732BindGlobal( "NextIterator_List", function ( iter )
733    if iter!.pos = Length( iter!.list ) then
734        Error("<iter> is exhausted");
735    fi;
736    iter!.pos := iter!.pos + 1;
737    while not IsBound( iter!.list[ iter!.pos ] ) do
738        iter!.pos := iter!.pos + 1;
739    od;
740    return iter!.list[ iter!.pos ];
741    end );
742
743#BindGlobal( "IsDoneIterator_DenseList",
744#    iter -> not IsBound( iter!.list[ iter!.pos + 1 ] ) );
745
746BindGlobal( "NextIterator_DenseList", function ( iter )
747    iter!.pos := iter!.pos + 1;
748    #if not IsBound( iter!.list[ iter!.pos ] ) then
749    #    Error("<iter> is exhausted");
750    #fi;
751    return iter!.list[ iter!.pos ];
752    end );
753
754BindGlobal( "ShallowCopy_List",
755    iter -> rec( list := iter!.list,
756                 pos  := iter!.pos,
757                 len  := iter!.len ) );
758
759InstallGlobalFunction( IteratorList, function ( list )
760    local   iter;
761
762    iter := rec(
763                list := list,
764#T call `Immutable'?
765                pos  := 0,
766                len := Length(list),
767                ShallowCopy := ShallowCopy_List );
768
769    if IsDenseList( list ) and not IsMutable( list ) then
770      iter.IsDoneIterator := IsDoneIterator_List;
771      iter.NextIterator   := NextIterator_DenseList;
772    else
773      iter.IsDoneIterator := IsDoneIterator_List;
774      iter.NextIterator   := NextIterator_List;
775    fi;
776
777    return IteratorByFunctions( iter );
778end );
