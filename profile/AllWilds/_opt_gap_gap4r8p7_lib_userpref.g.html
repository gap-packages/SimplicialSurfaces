28##  <Func Name="WriteGapIniFile" Arg="[dir][,][ignorecurrent]"/>

30##  <Description>
31##  
32##  Some aspects of the behaviour of &GAP; can be customized by the user via
33##  <Emph>user preferences</Emph>.  Examples include  the way  help sections
34##  are displayed or the use of colors in the terminal. <P/>
35##  
36##  User preferences are  specified via a pair of strings,  the first is the
37##  (case insensitive) name of a package (or <C>"GAP"</C> for the core &GAP;
38##  library) and the second is some arbitrary case sensitive string. <P/>
39##  
40##  User   preferences  can   be  set   to  some   <A>value</A>  with   <Ref
41##  Func="SetUserPreference"/>. The  current value of a  user preference can
42##  be found with <Ref Func="UserPreference"/>. In both cases, if no package
43##  name is  given the default  <C>"GAP"</C> is  used. If a  user preference
44##  is  not  known or  not  set  then <Ref  Func="UserPreference"/>  returns
45##  <K>fail</K>. <P/>
46##  
47##  The function <Ref Func="ShowUserPreferences"/> with no argument shows in
48##  a  pager  an  overview  of  all known  user  preferences  together  with
49##  some  explanation  and  the  current  value.  If  one  or  more  strings
50##  <A>package1</A>, ... are given then  only the user preferences for these
51##  packages are shown. <P/>
52##  
53##  The easiest way to  make use of user preferences is  probably to use the
54##  function <Ref  Func="WriteGapIniFile"/>, usually without  argument. This
55##  function creates a file <F>gap.ini</F>  in your user specific &GAP; root
56##  directory (<C>GAPInfo.UserGapRoot</C>).  If such  a file  already exists
57##  the function  will make a  backup of it  first. This newly  created file
58##  contains  descriptions of  all  known user  preferences  and also  calls
59##  of  <Ref Func="SetUserPreference"/>  for  those  user preferences  which
60##  currently do not  have their default value. You can  then edit that file
61##  to customize (further)  the user preferences for  future &GAP; sessions.
62##  <P/>
63##  
64##  Should a  later version  of &GAP;  or some  packages introduce  new user
65##  preferences then you can  call <Ref Func="WriteGapIniFile"/> again since
66##  it  will set  the previously  known  user preferences  to their  current
67##  values. <P/>
68##  
69##  Optionally,  a  different  directory for  the  resulting  <F>gap.ini</F>
70##  file    can   be    specified   as    argument   <A>dir</A>    to   <Ref
71##  Func="WriteGapIniFile"/>. Another optional argument is the boolean value
72##  <K>true</K>, if this  is given, the settings of all  user preferences in
73##  the current session are ignored. <P/>
74##  
75##  Note that  your <F>gap.ini</F> file is  read by &GAP; very  early during
76##  its startup process. A consequence  is that the <A>value</A> argument in
77##  a call of <Ref Func="SetUserPreference"/>  must be some very basic &GAP;
78##  object, usually a boolean, a number, a  string or a list of those. A few
79##  user  preferences support  more complicated  settings. For  example, the
80##  user  preference <C>"UseColorPrompt"</C>  admits a  record as  its value
81##  whose components are available  only after the <Package>GAPDoc</Package>
82##  package has been loaded, see&nbsp;<Ref Func="ColorPrompt"/>. If you want
83##  to specify such a complicated value, then move the corresponding call of
84##  <Ref Func="SetUserPreference"/> from your  <F>gap.ini</F> file into your
85##  <F>gaprc</F>  file (also  in the  directory <C>GAPInfo.UserGapRoot</C>).
86##  This file is read much later. <P/>
87##  
88##  <Example>
89##  gap> SetUserPreference( "Pager", "less" );
90##  gap> SetUserPreference("PagerOptions",
91##  >                      [ "-f", "-r", "-a", "-i", "-M", "-j2" ] );
92##  gap> UserPreference("Pager");
93##  "less"

98##  available on  your system. The  last line displays the  current setting.

107##  <Description>
108##  
109##  This  function can  be used  (also in  packages) to  introduce new  user
110##  preferences. It declares  a user preference, determines  a default value
111##  and contains documentation  of the user preference.  After declaration a
112##  user preference will be shown with <Ref Func="ShowUserPreferences"/> and
113##  <Ref Func="WriteGapIniFile"/>. <P/>
114##  
115##  When  this  declaration  is  evaluated  it  is  checked,  if  this  user
116##  preference is  already set in the  current session. If not  the value of
117##  the user  preference is set to  its default. (Do not  use <K>fail</K> as
118##  default  value  since this  indicated  that  a  user preference  is  not
119##  set.)<P/>
120##  
121##  The argument <A>record</A> of <Ref Func="DeclareUserPreference"/> must be
122##  a record with the following components.
123##  <P/>
124##  <List>
125##  <Mark><C>name</C></Mark>
126##  <Item>
127##    a string or a list of strings, the latter meaning several preferences
128##    which belong together,
129##  </Item>
130##  <Mark><C>description</C></Mark>
131##  <Item>
132##    a list of strings describing the preference(s), one string for each
133##    paragraph;
134##    if several preferences are declared together then the description
135##    refers to all of them,
136##  </Item>
137##  <Mark><C>default</C></Mark>
138##  <Item>
139##    the default value that is used,
140##    or a function without arguments that computes this default value;
141##    if several preferences are declared together then the value of this
142##    component must be the list of default values for the individual
143##    preferences.
144##  </Item>
145##  </List>
146##  <P/>
147##  The following components of <A>record</A> are optional.
148##  <P/>
149##  <List>
150##  <Mark><C>check</C></Mark>
151##  <Item>
152##    a function that takes a value as its argument and returns either
153##    <K>true</K> or <K>false</K>, depending on whether the given value
154##    is admissible for this preference;
155##    if several preferences are declared together then the number of
156##    arguments of the function must equal the length of the <C>name</C>
157##    list,
158##  </Item>
159##  <Mark><C>values</C></Mark>
160##  <Item>
161##    the list of admissible values, or a function without arguments
162##    that returns this list,
163##  </Item>
164##  <Mark><C>multi</C></Mark>
165##  <Item>
166##    <K>true</K> or <K>false</K>, depending on whether one may choose
167##    several values from the given list or just one;
168##    needed (and useful only) if the <C>values</C> component is present,
169##  </Item>
170##  <Mark><C>package</C></Mark>
171##  <Item>
172##    the name of the &GAP; package to which the preference is assigned;
173##    if the declaration happens inside a file that belongs to this package
174##    then the value of this component is computed,
175##    using <C>GAPInfo.PackageCurrent</C>;
176##    otherwise, the default value for <C>package</C> is <C>"GAP"</C>,
177##  </Item>
178##  <Mark><C>omitFromGapIniFile</C></Mark>
179##  <Item>
180##    if the value is <K>true</K> then this user preference is ignored by
181##    <Ref Func="WriteGapIniFile"/>.
182##  </Item>
183##  </List>
184##  <P/>

566StringUserPreferences:= function( arg )
567    local ignorecurrent, pref, str, pkglist, pkgname, done, name, data;
568    
569    if Length(arg) > 0 and arg[1] = true then
570      ignorecurrent := true;
571    else
572      ignorecurrent := false;
573    fi;
574
575    pref:= GAPInfo.UserPreferences;
576    str:= "";
577
578    # Run over the preferences, first the ones that belong to GAP,
579    # then the ones that belong to packages
580    pkglist := Concatenation(["gap"], Difference(RecNames( pref ), ["gap"] ));
581## HACKUSERPREF  temporary until all packages are adjusted
582    pkglist := Filtered(pkglist, a-> not a in ["Pager","ReadObsolete"]);
583    for pkgname in pkglist do
584      Append( str, ListWithIdenticalEntries( 77, '#' ) );
585      Append( str, "\n\n" );
586      done:= [];
587      if IsRecord( pref.( pkgname ) ) then
588        for name in Set( RecNames( pref.( pkgname ) ) ) do
589          if not name in done then
590            data:= DataOfUserPreference( pkgname, name );
591            if not IsBound( data.undeclared ) and
592               not data.omitFromGapIniFile then
593              Append( str, StringUserPreference( data, ignorecurrent ) );
594              Append( str, "\n" );
595            fi;
596            UniteSet( done, data.names );
597          fi;
598        od;
599      fi;
600    od;
601
602    return str;
603    end;

611##
612ShowUserPreferences:= function(arg)

614

619      # if no list given use all  packages with preferences, "gap" first
620      pkglist := Concatenation(  [ "gap" ], 
621                       Difference( RecNames( pref ), [ "gap" ] ) );
622    fi;
623
624## HACKUSERPREF  temporary until all packages are adjusted
625    pkglist := Filtered(pkglist, a-> not a in ["Pager","ReadObsolete"]);
626
627    str:= "";

629      Append( str, "\n" );

632        nam := GAPInfo.PackagesInfo.(pkgname)[1].PackageName;

635      else 
636        nam := pkgname;
637      fi;

674##
675#F  WriteGapIniFile( [<dir>, ][true] )
676##
677WriteGapIniFile:= function( arg )
678  local ignorecurrent, f, df, ret, target, str, res;
679  # check if current settings should be used
680  if true in arg then 
681    ignorecurrent := true;
682  else 
683    ignorecurrent := false;
684  fi;
685  # check if target directory is given as string or directory object
686  f := First(arg, IsString);
687  if f <> fail then
688    df := Directory(f);
689  else
690    df := First(arg, IsDirectory);
691    if df <> fail then
692      f := df![1];
693    fi;
694  fi;
695  # otherwise use users GAPInfo.UserGapRoot
696  if f = fail then
697    if GAPInfo.UserGapRoot = fail then
698      Error("Your system does not support a user specific default root path.\n\
699Please specify directory to write gap.ini file.");
700      return fail;
701    else
702      f := GAPInfo.UserGapRoot;
703      df := Directory(f);
704    fi;
705  fi;
706  # maybe create GAPInfo.UserGapRoot
707  if not IsDirectoryPath(f) then
708    ret := CreateDir(f);
709    if ret = fail then
710      Error("Cannot create directory ",f,"\nError message: ",
711            LastSystemError().message,"\n");
712      return fail;
713    fi;
714  fi;
715  # name of target file
716  target := Filename(df, "gap.ini");
717  # if target exists copy it to <target>.bak
718  if IsExistingFile(target) then
719    str := StringFile(target);
720    if str = fail then
721      Error("Cannot read existing file ",target,".\n");
722      return fail;
723    fi;
724    ret := FileString(Concatenation(target, ".bak"), str);
725    if ret = fail then
726      Error("Cannot write backup file ",target,".bak.\nError message: ",
727            LastSystemError().message,".\n");
728      return fail;
729    else
730      Info(InfoWarning, 1, "Copied existing gap.ini to ", target, ".bak");
731    fi;
732  fi;
733  # content of resulting file as string
734  res := StringUserPreferences(ignorecurrent);
735  if ARCH_IS_WINDOWS() then
736    # use DOS/Windows style line breaks
737    res := ReplacedString(res, "\n", "\r\n");
738  fi;
739  # finally write the gap.ini file
740  ret := FileString(target, res);
741  if ret = fail then
742    Error("Cannot write target file ",target,".\nError message: ",
743          LastSystemError().message,".\n");
744    return fail;
745  fi;
746  Info(InfoWarning, 1, "File ", target, " successfully written.");
747  return true;
748end;
